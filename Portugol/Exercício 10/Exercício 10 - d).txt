programa
{
  funcao inicio()
  {  
    real salario 
    inteiro ano
    cadeia nome
    caracter ala

    escreva("Digite o seu nome: ")
    leia(nome)
    escreva("Digite o seu salário: ")
    leia(salario)
    escreva("Digite o tempo em que trabalhou na empresa(em anos): ")
    leia(ano)
    escreva("Você trabalha na ala operacional(O) ou administrativa(A) da empresa (responda com O ou A): ")
    leia(ala)

    se(ano <= 3)
    {
      salario = salario * 1.03 
    }
    se(ano > 3 e ano < 10)
    {
      salario = salario * 1.125
    }
    se(ano >= 10)
    {
      salario = salario * 1.2
    }
    se(ala == "O")
    {
      salario = salario * 1.05
    }
    escreva("\nSeu recebimento mensal será de: ", salario," reais")
  }
}
